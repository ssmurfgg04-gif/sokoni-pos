    76→function t(locale: Locale, path: string): string {
    77→  const keys = path.split('.');
    78→  let current: any = translations[locale];
    79→  for (const key of keys) {
    80→    if (current && typeof current === 'object' && key in current) { current = current[key]; }
    81→    else { let fb: any = translations.en; for (const k of keys) { if (fb && typeof fb === 'object' && k in fb) { fb = fb[k]; } else return path; } return typeof fb === 'string' ? fb : path; }
    82→  }
    83→  return typeof current === 'string' ? current : path;
    84→}

   110→// MINI BAR CHART (pure CSS, no library needed)
   111→// ============================================
   112→function MiniBarChart({ data, maxValue, color = 'bg-primary' }: { data: number[]; maxValue: number; color?: string }) {
   113→  return (
   114→    <div className="flex items-end gap-1 h-16">
   115→      {data.map((val, i) => {
   116→        const height = maxValue > 0 ? Math.max((val / maxValue) * 100, 2) : 2;
   117→        return (
   118→          <div key={i} className="flex-1 flex flex-col items-center gap-1">
   119→            <div className="w-full relative group">
   120→              <div className={`w-full ${color} rounded-t-sm transition-all duration-500`} style={{ height: `${height}px` }} />
   121→              <div className="absolute -top-6 left-1/2 -translate-x-1/2 bg-foreground text-background text-[10px] px-1.5 py-0.5 rounded opacity-0 group-hover:opacity-100 transition-opacity whitespace-nowrap pointer-events-none">
   122→                {formatCurrency(val)}
   123→              </div>
   124→            </div>
   125→          </div>
   126→        );
   127→      })}
   128→    </div>
   129→  );
   130→}

   132→// ============================================
   133→// PROGRESS RING (for compliance score)
   134→// ============================================
   135→function ProgressRing({ value, size = 80, strokeWidth = 6, color }: { value: number; size?: number; strokeWidth?: number; color: string }) {
   136→  const radius = (size - strokeWidth) / 2;
   137→  const circumference = radius * 2 * Math.PI;
   138→  const offset = circumference - (value / 100) * circumference;
   139→  return (
   140→    <svg width={size} height={size} className="-rotate-90">
   141→      <circle cx={size / 2} cy={size / 2} r={radius} fill="none" stroke="currentColor" strokeWidth={strokeWidth} className="text-muted/30" />
   142→      <circle cx={size / 2} cy={size / 2} r={radius} fill="none" stroke={color} strokeWidth={strokeWidth}
   143→        strokeDasharray={circumference} strokeDashoffset={offset} strokeLinecap="round" className="transition-all duration-1000" />
   144→    </svg>
   145→  );
   146→}
